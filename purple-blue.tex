		 on line,
		 frame empty,
		 coltext=InlineCodeBgColor,
		 coltext=CodeColor,
		 boxsep=0.02083334in,
		 outer arc=0.3em,
		 arc=0.3em,
		 top=0.02083334in,
		 right=0.04166668in,
		 bottom=0.02083334in,
		 left=0.04166668in,
		 fontlower=\codesize
	]{
		#1
	}
}
